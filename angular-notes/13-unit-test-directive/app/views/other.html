过滤器filter的使用

{{ '\{\{ string \}\}' }} :

{{ string }}

{{ '\{\{ string | limitTo : 10 \}\}' }}:

{{ string | limitTo : 10 }}

{{ '\{\{ string | lowercase \}\}' }}:

{{ string | lowercase }}

{{ '\{\{ string | uppercase \}\}' }}:

{{ string | uppercase }}

{{ '\{\{ date | date : \'yyyy-MM-dd hh:mm:ss\' \}\}' }}:

{{ date | date : 'yyyy-MM-dd hh:mm:ss' }}

{{ '\{\{ date | date : \'yyyy-MM-dd\' | mydate \}\}' }}:

{{ date | date : 'yyyy-MM-dd' | mydate }}

{{ '\{\{ date | date : \'yyyy-MM-dd\' | mydate : \'cn\' \}\}' }}:

{{ date | date : 'yyyy-MM-dd' | mydate : 'cn' }}

{{ '\{\{ number \}\}' }}:

{{ number }}

{{ '\{\{ number | number \}\}' }}:

{{ number | number }}

{{ '\{\{ number_with_currency \}\}' }}:

{{ number_with_currency }}
